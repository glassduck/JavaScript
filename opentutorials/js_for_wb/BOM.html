console.log(location); //이러면 location이 가지고 있는 모든 속성을 보여줌
alert(location) //alert는 string값만 받기 때문에 자동적으로 locaiton.toString()이 들어간다.
console.log(location.protocol, location.host, location.port, location.pathname, location.search, location.hash)
//http: protocol ,사이트 주소 :host, port, 특정한 정보 : pathname, ?뒤에 붙는 전달한 값 : search
//,문서 내 특정한 위치(북마크) : hash.
